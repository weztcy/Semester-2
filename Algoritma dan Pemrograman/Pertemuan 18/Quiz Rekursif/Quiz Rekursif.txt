#include <iostream>

using namespace std;

float rec_mean (int data[], int cur, int n)
{
    if (cur == n-1)
        return data[cur];
    if (cur == 0)
        return ((data[cur] + rec_mean(data, cur+1, n))/n);
 
    return (data[cur] + rec_mean(data, cur+1, n));
}
int main()
{
    int data1[] = {12, 15, 1, 23, 44, 21, 30};

    int size_data1 = sizeof(data1) / sizeof(data1[0]);

    cout << rec_mean(data1,0,size_data1) << endl;
    return 0;
}

#include <iostream>

using namespace std;

void rec_print_all(int data[], int cur, int arrSize)
{
    if (cur == arrSize)
    {
        return;
    }
    else
    {
        cout << data[cur] << " ";
        rec_print_all(data, cur + 1, arrSize);
    }
}

int main()
{
    int data1[] = {12, 15, 1, 23, 44, 21, 30};
    int size_data1 = sizeof(data1) / sizeof(data1[0]);
    rec_print_all(data1, 0, size_data1);
    return 0;
}

int max2(int num1, int num2)
{
    if(num1 > num2)
    {
        return num1;
    }
    else{
        return num2;
    }
}
int rec_max(int data[], int cur, int arrSize)
{
    if(cur==arrSize)
    {
        return data[0];
    }
    else
    {
        return max2(data[cur], rec_max(data, cur+1, arrSize));
